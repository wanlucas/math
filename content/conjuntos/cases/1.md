```
 A = { x | x é um número inteiro }

 -4 ∈ A, 1/3 ∉ A, 3 ∈ A e 0,25 ∉ A
```

```
 B = { x | x < 1 }

```

```

```

## 3 - Dado H = { -1, 0, 2, 4, 9 }

```
 A = { x | x ∈ H e x < 1 }
 A = { -1, 0 }
```

```
 B = { x | x ∈ H e (2x - 1) / 3 = 1 }
 B = { x | x ∈ H e 2x - 1 = 3 }
 B = { x | x ∈ H e x = 2 }
 B = { 2 }
```

```
 C = { x | x ∈ H e x é quadrado perfeito }
 C = { 0, 4, 9 }
```

```
 D = { x | x ∈ H e 3x + 1 = 10 }
 D = { x | x ∈ H e x = 3 }
 D = ∅
```
